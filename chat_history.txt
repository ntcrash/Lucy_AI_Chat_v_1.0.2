```
import requests

# Set your own api key and endpoint url
API_KEY = "your_api_key"
ENDPOINT_URL = "https://api.gpt4all.org/v1/models/Meta-LLaMA-3-8B-Instruct.Q4_0"

def send_request(prompt, max_tokens):
    headers = {"Authorization": f"Bearer {API_KEY}"}
    data = {
        "prompt": prompt,
        "max_tokens": str(max_tokens),
        "temperature": 1.0
    }
    response = requests.post(ENDPOINT_URL, json=data, headers=headers)
    return response.json()

def main():
    # Set your own prompts and max tokens here
    prompt = "Write a short story about..."
    max_tokens = 100

    response = send_request(prompt, max_tokens)

    if response["status"] == "success":
        print("Response:")
        for text in response["text"]:
            print(text)
    else:
        print(f"Error: {response['error']}")

if __name__ == "__main__":
    main()
```
